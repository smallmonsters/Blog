一、使用sass

> yarn add node - sadd sass - loader--dev

``` JavaScript
// vue.config.js
module.exports = {
    css: {
        loaderOptions: {
            scss: {
                prependData: `@import "global/scss/path";` 
            }
        }
    }
}
```

二、使用路径别名

``` JavaScript
// vue.config.js
module.exports = {
    chainWebpack: (config) => {
        config.resolve.alias
            .set("@",resolve("src"))
    }
}
```
